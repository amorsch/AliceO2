47 //< crossing coordinates of 2 circles
48 struct CircleCrossInfo {
49  float xDCA[2];
50  float yDCA[2];
51  int nDCA;
52 
53  CircleCrossInfo() = default;
54 
55  CircleCrossInfo(const TrackAuxPar& trc0, const TrackAuxPar& trc1) { set(trc0, trc1); }
56  int set(const TrackAuxPar& trc0, const TrackAuxPar& trc1)
57  {
58  // calculate up to 2 crossings between 2 circles
59  nDCA = 0;
60  const auto& trcA = trc0.rC > trc1.rC ? trc0 : trc1; // designate the largest circle as A
61  const auto& trcB = trc0.rC > trc1.rC ? trc1 : trc0;
62  float xDist = trcB.xC - trcA.xC, yDist = trcB.yC - trcA.yC;
63  float dist2 = xDist * xDist + yDist * yDist, dist = std::sqrt(dist2), rsum = trcA.rC + trcB.rC;
64  if (std::abs(dist) < 1e-12) {
65  return nDCA; // circles are concentric?
66  }
67  if (dist > rsum) { // circles don't touch, chose a point in between
68  // the parametric equation of lines connecting the centers is
69  // x = x0 + t/dist * (x1-x0), y = y0 + t/dist * (y1-y0)
70  notTouchingXY(dist, xDist, yDist, trcA, trcB.rC);
71  } else if (dist + trcB.rC < trcA.rC) { // the small circle is nestled into large one w/o touching
72  // select the point of closest approach of 2 circles
73  notTouchingXY(dist, xDist, yDist, trcA, -trcB.rC);
74  } else { // 2 intersection points
75  // to simplify calculations, we move to new frame x->x+Xc0, y->y+Yc0, so that
76  // the 1st one is centered in origin
77  if (std::abs(xDist) < std::abs(yDist)) {
78  float a = (trcA.rC * trcA.rC - trcB.rC * trcB.rC + dist2) / (2. * yDist), b = -xDist / yDist, ab = a * b, bb = b * b;
79  float det = ab * ab - (1. + bb) * (a * a - trcA.rC * trcA.rC);
80  if (det > 0.) {
81  det = std::sqrt(det);
82  xDCA[0] = (-ab + det) / (1. + b * b);
83  yDCA[0] = a + b * xDCA[0] + trcA.yC;
84  xDCA[0] += trcA.xC;
85  xDCA[1] = (-ab - det) / (1. + b * b);
86  yDCA[1] = a + b * xDCA[1] + trcA.yC;
87  xDCA[1] += trcA.xC;
88  nDCA = 2;
89  } else { // due to the finite precision the det<=0, i.e. the circles are barely touching, fall back to this special case
90  notTouchingXY(dist, xDist, yDist, trcA, trcB.rC);
91  }
92  } else {
93  float a = (trcA.rC * trcA.rC - trcB.rC * trcB.rC + dist2) / (2. * xDist), b = -yDist / xDist, ab = a * b, bb = b * b;
94  float det = ab * ab - (1. + bb) * (a * a - trcA.rC * trcA.rC);
95  if (det > 0.) {
96  det = std::sqrt(det);
97  yDCA[0] = (-ab + det) / (1. + bb);
98  xDCA[0] = a + b * yDCA[0] + trcA.xC;
99  yDCA[0] += trcA.yC;
100  yDCA[1] = (-ab - det) / (1. + bb);
101  xDCA[1] = a + b * yDCA[1] + trcA.xC;
102  yDCA[1] += trcA.yC;
103  nDCA = 2;
104  } else { // due to the finite precision the det<=0, i.e. the circles are barely touching, fall back to this special case
105  notTouchingXY(dist, xDist, yDist, trcA, trcB.rC);
106  }
107  }
108  }
109  return nDCA;
110  }

112  void notTouchingXY(float dist, float xDist, float yDist, const TrackAuxPar& trcA, float rBSign)
113  {
114  // fast method to calculate DCA between 2 circles, assuming that they don't touch each outer:
115  // the parametric equation of lines connecting the centers is x = xA + t/dist * xDist, y = yA + t/dist * yDist
116  // with xA,yY being the center of the circle A ( = trcA.xC, trcA.yC ), xDist = trcB.xC = trcA.xC ...
117  // There are 2 special cases:
118  // (a) small circle is inside the large one: provide rBSign as -trcB.rC
119  // (b) circle are side by side: provide rBSign as trcB.rC
120  nDCA = 1;
121  auto t2d = (dist + trcA.rC - rBSign) / dist;
122  xDCA[0] = trcA.xC + 0.5 * (xDist * t2d);
123  yDCA[0] = trcA.yC + 0.5 * (yDist * t2d);
124  }
125 };
