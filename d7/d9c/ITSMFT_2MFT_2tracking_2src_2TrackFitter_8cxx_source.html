110 void TrackFitter::initTrack(const o2::itsmft::Cluster& cl, TrackParamMFT& param)
111 {
112 
117 
118  // compute the track parameters at the last cluster
119 
120  //double k = -mBZField * o2::constants::math::B2C;
121  double x0 = cl.getX();
122  double y0 = cl.getY();
123  double dZ = -(cl.getZ());
124  double pt = TMath::Sqrt(x0 * x0 + y0 * y0) * 10;
125  double pz = -dZ * 10;
126  //double phi1 = TMath::ATan2(cl1.getY(), cl1.getX());
127  double phi2 = TMath::ATan2(cl.getY(), cl.getX());
128  double tanl = pz / pt;

130  double r0cu = r0sq * TMath::Sqrt(r0sq);
131  double sigmax0sq = 0.0005; // FIXME: from cluster
132  double sigmay0sq = 0.0005; // FIXME: from cluster
133  double sigmaDeltaZsq = 5; // Primary vertex distribution: beam interaction diamond
134  double sigmaboost = 1e7; // Boost q/pt seed covariances
135  double sigmainvqptsq = sigmaboost * 1e-8 / r0cu / r0cu * (sigmax0sq * x0 * x0 + sigmay0sq * y0 * y0); // .25 / pt; // Very large uncertainty on pt

137  std::cout << "initTrack: pt = " << pt << std::endl;
138  std::cout << " -iniTrack: cluster X = " << cl.getX() << " Y = " << cl.getY() << " Z = " << cl.getZ() << " phi2 = " << phi2 << std::endl;

143  param.setY(cl.getY());
144  param.setZ(cl.getZ());

147  param.setTanl(tanl);
148  std::cout << " seed Phi, Tanl, InvQpt = " << param.getPhi() << " " << param.getTanl() << " " << param.getInvQPt() << std::endl;
149 
150  // compute the track parameter covariances at the last cluster (as if the other clusters did not exist)
151  TMatrixD lastParamCov(5, 5);
152  lastParamCov.Zero();
153  lastParamCov(0, 0) = sigmax0sq; // <X,X>
154  lastParamCov(0, 1) = 0; // <Y,X>
155  lastParamCov(0, 2) = -sigmax0sq * y0 / r0sq; // <PHI,X>
156  lastParamCov(0, 3) = -dZ * sigmax0sq * x0 / r0cu; // <TANL,X>
157  lastParamCov(0, 4) = sigmaboost * -1e-4 * x0 * sigmax0sq / r0cu; // <INVQPT,X>
158 
159  lastParamCov(1, 1) = sigmay0sq; // <Y,Y>

163 
164  lastParamCov(2, 2) = (sigmax0sq * y0 * y0 + sigmay0sq * x0 * x0) / r0sq / r0sq; // <PHI,PHI>
165  lastParamCov(2, 3) = dZ * x0 * y0 * (sigmax0sq - sigmay0sq) / r0sq / r0cu + 1e-10; // <TANL,PHI>
166  lastParamCov(2, 4) = sigmaboost * 1e-4 * y0 * x0 / r0cu / r0sq * (sigmax0sq - sigmay0sq) + 1e-10; // <INVQPT,PHI>
167 
168  lastParamCov(3, 3) = dZ * dZ * (sigmax0sq * x0 * x0 + sigmay0sq * y0 * y0) / r0cu / r0cu + sigmaDeltaZsq / r0sq; // <TANL,TANL>
169  lastParamCov(3, 4) = sigmaboost * 1e-4 * dZ / r0cu / r0cu * (sigmax0sq * x0 * x0 + sigmay0sq * y0 * y0); // <INVQPT,TANL>
170 
171  lastParamCov(4, 4) = sigmainvqptsq; // <INVQPT,INVQPT>
172 
173  lastParamCov(1, 0) = lastParamCov(0, 1); //
174  lastParamCov(2, 0) = lastParamCov(0, 2); //
175  lastParamCov(2, 1) = lastParamCov(1, 2); //

177  lastParamCov(3, 1) = lastParamCov(1, 3); //
178  lastParamCov(3, 2) = lastParamCov(2, 3); //
179  lastParamCov(4, 0) = lastParamCov(0, 4); //
180  lastParamCov(4, 1) = lastParamCov(1, 4); //
181  lastParamCov(4, 2) = lastParamCov(2, 4); //

265 void TrackFitter::smoothTrack(FitterTrackMFT& track, bool finalize)
266 {
271 
272  auto itCurrentParam(track.begin());
273  auto itPreviousParam(itCurrentParam);
274  ++itCurrentParam;
275 
276  // smoothed parameters and covariances at first cluster = filtered parameters and covariances
277  itPreviousParam->setSmoothParameters(itPreviousParam->getParameters());
278  itPreviousParam->setSmoothCovariances(itPreviousParam->getCovariances());
279 
280  // local chi2 at first cluster = last additional chi2 provided by Kalman
281  itPreviousParam->setLocalChi2(itPreviousParam->getTrackChi2() - itCurrentParam->getTrackChi2());
282 
283  // recursively smooth the next parameters and covariances
284  do {
285  try {
286  runSmoother(*itPreviousParam, *itCurrentParam);
287  } catch (std::exception const&) {
288  throw;
289  }
290  ++itPreviousParam;
291  } while (++itCurrentParam != track.end());
292 
293  // update the regular parameters and covariances if requested
294  if (finalize) {
295  for (auto& param : track) {
296  param.setParameters(param.getSmoothParameters());
297  param.setCovariances(param.getSmoothCovariances());
298  }
299  }
300 }

303 void TrackFitter::runKalmanFilter(TrackParamMFT& trackParam)
304 {
308 
309  // get actual track parameters (p)
310  TMatrixD param(trackParam.getParameters());
311 
312  // get new cluster parameters (m)
313  const o2::itsmft::Cluster* cluster = trackParam.getClusterPtr();
314  TMatrixD clusterParam(5, 1);
315  clusterParam.Zero();
316  clusterParam(0, 0) = cluster->getX();
317  clusterParam(1, 0) = cluster->getY();
318 
319  // compute the actual parameter weight (W)
320  TMatrixD paramWeight(trackParam.getCovariances());
321  if (paramWeight.Determinant() != 0) {
322  paramWeight.Invert();
323  } else {
324  throw std::runtime_error("Determinant = 0");
325  }
326 
327  // compute the new cluster weight (U)
328  TMatrixD clusterWeight(5, 5);
329  clusterWeight.Zero();
330  clusterWeight(0, 0) = 1. / cluster->getSigmaZ2(); // 1. / cluster->getEx2();
331  clusterWeight(1, 1) = 1. / cluster->getSigmaY2(); // 1. / cluster->getEy2();
332 
333  // compute the new parameters covariance matrix ((W+U)^-1)
334  TMatrixD newParamCov(paramWeight, TMatrixD::kPlus, clusterWeight);
335  if (newParamCov.Determinant() != 0) {
336  newParamCov.Invert();
337  } else {
338  throw std::runtime_error("Determinant = 0");
339  }
340  trackParam.setCovariances(newParamCov);
341 
342  // compute the new parameters (p' = ((W+U)^-1)U(m-p) + p)
343  TMatrixD tmp(clusterParam, TMatrixD::kMinus, param); // m-p
344  TMatrixD tmp2(clusterWeight, TMatrixD::kMult, tmp); // U(m-p)
345  TMatrixD newParam(newParamCov, TMatrixD::kMult, tmp2); // ((W+U)^-1)U(m-p)
346  newParam += param; // ((W+U)^-1)U(m-p) + p
347  trackParam.setParameters(newParam);
348 
349  // compute the additional chi2 (= ((p'-p)^-1)W(p'-p) + ((p'-m)^-1)U(p'-m))
350  tmp = newParam; // p'
351  tmp -= param; // (p'-p)
352  TMatrixD tmp3(paramWeight, TMatrixD::kMult, tmp); // W(p'-p)
353  TMatrixD addChi2Track(tmp, TMatrixD::kTransposeMult, tmp3); // ((p'-p)^-1)W(p'-p)
354  tmp = newParam; // p'
355  tmp -= clusterParam; // (p'-m)
356  TMatrixD tmp4(clusterWeight, TMatrixD::kMult, tmp); // U(p'-m)
357  addChi2Track += TMatrixD(tmp, TMatrixD::kTransposeMult, tmp4); // ((p'-p)^-1)W(p'-p) + ((p'-m)^-1)U(p'-m)
358  trackParam.setTrackChi2(trackParam.getTrackChi2() + addChi2Track(0, 0));
359 }

362 void TrackFitter::runSmoother(const TrackParamMFT& previousParam, TrackParamMFT& param)
363 {
366 
367  // get variables
368  const TMatrixD& extrapParameters = previousParam.getExtrapParameters(); // X(k+1 k)
369  const TMatrixD& filteredParameters = param.getParameters(); // X(k k)
370  const TMatrixD& previousSmoothParameters = previousParam.getSmoothParameters(); // X(k+1 n)
371  const TMatrixD& propagator = previousParam.getPropagator(); // F(k)
372  const TMatrixD& extrapCovariances = previousParam.getExtrapCovariances(); // C(k+1 k)
373  const TMatrixD& filteredCovariances = param.getCovariances(); // C(k k)
374  const TMatrixD& previousSmoothCovariances = previousParam.getSmoothCovariances(); // C(k+1 n)
375 
376  // compute smoother gain: A(k) = C(kk) * F(k)^t * (C(k+1 k))^-1
377  TMatrixD extrapWeight(extrapCovariances);
378  if (extrapWeight.Determinant() != 0) {
379  extrapWeight.Invert(); // (C(k+1 k))^-1
380  } else {
381  throw std::runtime_error("Determinant = 0");
382  }
383  TMatrixD smootherGain(filteredCovariances, TMatrixD::kMultTranspose, propagator); // C(kk) * F(k)^t
384  smootherGain *= extrapWeight; // C(kk) * F(k)^t * (C(k+1 k))^-1
385 
386  // compute smoothed parameters: X(k n) = X(k k) + A(k) * (X(k+1 n) - X(k+1 k))
387  TMatrixD tmpParam(previousSmoothParameters, TMatrixD::kMinus, extrapParameters); // X(k+1 n) - X(k+1 k)
388  TMatrixD smoothParameters(smootherGain, TMatrixD::kMult, tmpParam); // A(k) * (X(k+1 n) - X(k+1 k))
389  smoothParameters += filteredParameters; // X(k k) + A(k) * (X(k+1 n) - X(k+1 k))
390  param.setSmoothParameters(smoothParameters);
391 
392  // compute smoothed covariances: C(k n) = C(k k) + A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
393  TMatrixD tmpCov(previousSmoothCovariances, TMatrixD::kMinus, extrapCovariances); // C(k+1 n) - C(k+1 k)
394  TMatrixD tmpCov2(tmpCov, TMatrixD::kMultTranspose, smootherGain); // (C(k+1 n) - C(k+1 k)) * (A(k))^t
395  TMatrixD smoothCovariances(smootherGain, TMatrixD::kMult, tmpCov2); // A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
396  smoothCovariances += filteredCovariances; // C(k k) + A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
397  param.setSmoothCovariances(smoothCovariances);
398 
399  // compute smoothed residual: r(k n) = cluster - X(k n)
400  const o2::itsmft::Cluster* cluster = param.getClusterPtr();
401  TMatrixD smoothResidual(2, 1);
402  smoothResidual.Zero();
403  smoothResidual(0, 0) = cluster->getX() - smoothParameters(0, 0);
404  smoothResidual(1, 0) = cluster->getY() - smoothParameters(1, 0);
405 
406  // compute weight of smoothed residual: W(k n) = (clusterCov - C(k n))^-1
407  TMatrixD smoothResidualWeight(2, 2);
408  smoothResidualWeight(0, 0) = cluster->getSigmaZ2() - smoothCovariances(0, 0); // cluster->getEx2() - smoothCovariances(0, 0);
409  smoothResidualWeight(0, 1) = -smoothCovariances(0, 2);
410  smoothResidualWeight(1, 0) = -smoothCovariances(2, 0);
411  smoothResidualWeight(1, 1) = cluster->getSigmaY2() - smoothCovariances(2, 2); // cluster->getEy2() - smoothCovariances(2, 2);
412  if (smoothResidualWeight.Determinant() != 0) {
413  smoothResidualWeight.Invert();
414  } else {
415  throw std::runtime_error("Determinant = 0");
416  }
417 
418  // compute local chi2 = (r(k n))^t * W(k n) * r(k n)
419  TMatrixD tmpChi2(smoothResidual, TMatrixD::kTransposeMult, smoothResidualWeight); // (r(k n))^t * W(k n)
420  TMatrixD localChi2(tmpChi2, TMatrixD::kMult, smoothResidual); // (r(k n))^t * W(k n) * r(k n)
421  param.setLocalChi2(localChi2(0, 0));
422 }
