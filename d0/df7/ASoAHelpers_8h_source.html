65 template <typename T2, typename ARRAY, typename T>
66 std::vector<std::pair<uint64_t, uint64_t>> doGroupTable(const T& inTable, const std::string& categoryColumnName, int minCatSize)
67 {
68  auto arrowTable = inTable.asArrowTable();
69  auto columnIndex = arrowTable->schema()->GetFieldIndex(categoryColumnName);
70  auto chunkedArray = arrowTable->column(columnIndex)->data();
71 
72  uint64_t ind = 0;
73  std::vector<std::pair<uint64_t, uint64_t>> groupedIndices;
74  groupedIndices.reserve(arrowTable->num_rows());
75  for (uint64_t ci = 0; ci < chunkedArray->num_chunks(); ++ci) {
76  auto chunk = chunkedArray->chunk(ci);
77  // Note: assuming that the table is not empty
78  T2 const* data = std::static_pointer_cast<ARRAY>(chunk)->raw_values();
79  for (uint64_t ai = 0; ai < chunk->length(); ++ai) {
80  groupedIndices.emplace_back(data[ai], ind);
81  ind++;
82  }
83  }
84 
85  // Do a stable sort so that same categories entries are
86  // grouped together.
87  std::stable_sort(groupedIndices.begin(), groupedIndices.end());
88 
89  // Remove categories of too small size
90  if (minCatSize > 1) {
91  auto catBegin = groupedIndices.begin();
92  while (catBegin != groupedIndices.end()) {
93  auto catEnd = std::upper_bound(catBegin, groupedIndices.end(), *catBegin, sameCategory);
94  if (std::distance(catBegin, catEnd) < minCatSize) {
95  catEnd = groupedIndices.erase(catBegin, catEnd);
96  }
97  catBegin = catEnd;
98  }
99  }
100 
101  return groupedIndices;
102 }

104 template <typename T>
105 auto groupTable(const T& inTable, const std::string& categoryColumnName, int minCatSize)
106 {
107  auto arrowTable = inTable.asArrowTable();
108  auto columnIndex = arrowTable->schema()->GetFieldIndex(categoryColumnName);
109  auto dataType = arrowTable->column(columnIndex)->type();
110  if (dataType->id() == arrow::Type::UINT64) {
111  return doGroupTable<uint64_t, arrow::UInt64Array>(inTable, categoryColumnName, minCatSize);
112  }
113  if (dataType->id() == arrow::Type::INT64) {
114  return doGroupTable<int64_t, arrow::Int64Array>(inTable, categoryColumnName, minCatSize);
115  }
116  if (dataType->id() == arrow::Type::UINT32) {
117  return doGroupTable<uint32_t, arrow::UInt32Array>(inTable, categoryColumnName, minCatSize);
118  }
119  if (dataType->id() == arrow::Type::INT32) {
120  return doGroupTable<int32_t, arrow::Int32Array>(inTable, categoryColumnName, minCatSize);
121  }
122  // FIXME: Should we support other types as well?
123  throw std::runtime_error("Combinations: category column must be of integral type");
124 }

126 // Synchronize categories so as groupedIndices contain elements only of categories common to all tables
127 template <std::size_t K>
128 void syncCategories(std::array<std::vector<std::pair<uint64_t, uint64_t>>, K>& groupedIndices)
129 {
130  std::vector<std::pair<uint64_t, uint64_t>> firstCategories;
131  std::vector<std::pair<uint64_t, uint64_t>> commonCategories;
132  std::unique_copy(groupedIndices[0].begin(), groupedIndices[0].end(), std::back_inserter(firstCategories), sameCategory);
133 
134  for (auto& cat : firstCategories) {
135  bool isCommon = true;
136  for (int i = 1; i < K; i++) {
137  if (!std::binary_search(groupedIndices[i].begin(), groupedIndices[i].end(), cat, sameCategory)) {
138  isCommon = false;
139  break;
140  }
141  }
142  if (isCommon) {
143  commonCategories.push_back(cat);
144  }
145  }
146 
147  for (int i = 0; i < K; i++) {
148  auto nonCatBegin = groupedIndices[i].begin();
149  for (auto& cat : commonCategories) {
150  auto nonCatEnd = std::lower_bound(nonCatBegin, groupedIndices[i].end(), cat, sameCategory);
151  nonCatEnd = groupedIndices[i].erase(nonCatBegin, nonCatEnd);
152  nonCatBegin = std::upper_bound(nonCatEnd, groupedIndices[i].end(), cat, sameCategory);
153  }
154  }
155 }

179 template <typename... Ts>
180 struct CombinationsUpperIndexPolicy : public CombinationsIndexPolicyBase<Ts...> {
181  CombinationsUpperIndexPolicy(const Ts&... tables) : CombinationsIndexPolicyBase<Ts...>(tables...) {}
182 
183  void moveToEnd()
184  {
185  constexpr auto k = sizeof...(Ts);
186  for_<k>([&, this](auto i) {
187  std::get<i.value>(this->mCurrent).moveToEnd();
188  });
189  this->mIsEnd = true;
190  }
191 
192  void addOne()
193  {
194  constexpr auto k = sizeof...(Ts);
195  bool modify = true;
196  for_<k>([&, this](auto i) {
197  if (modify) {
198  constexpr auto curInd = k - i.value - 1;
199  std::get<curInd>(this->mCurrent)++;
200  if (*std::get<1>(std::get<curInd>(this->mCurrent).getIndices()) != std::get<curInd>(this->mMaxOffset)) {
201  for_<i.value>([&, this](auto j) {
202  constexpr auto curJ = k - i.value + j.value;
203  std::get<curJ>(this->mCurrent).setCursor(*std::get<1>(std::get<curJ - 1>(this->mCurrent).getIndices()));
204  });
205  modify = false;
206  }
207  }
208  });
209  this->mIsEnd = modify;
210  }
211 };

293 template <typename... Ts>
294 struct CombinationsBlockUpperIndexPolicy : public CombinationsIndexPolicyBase<Ts...> {
295  using IndicesType = typename generate_tuple_type<uint64_t, sizeof...(Ts)>::type;
296 
297  CombinationsBlockUpperIndexPolicy(const std::string& categoryColumnName, const Ts&... tables) : CombinationsIndexPolicyBase<Ts...>(tables...)

299  constexpr auto k = sizeof...(Ts);
300  if (this->mIsEnd) {
301  return;
302  }
303 
304  int tableIndex = 0;
305  ((this->mGroupedIndices[tableIndex++] = groupTable(tables, categoryColumnName, 1)), ...);
306 
307  // Synchronize categories across tables
308  // FIXME: This wouldn't be necessary if tables were always the same, are they?
309  syncCategories(this->mGroupedIndices);

319  {
320  constexpr auto k = sizeof...(Ts);
321  for_<k>([&, this](auto i) {
322  auto catBegin = this->mGroupedIndices[i.value].begin() + std::get<i.value>(this->mCurrentIndices);
323  auto range = std::equal_range(catBegin, this->mGroupedIndices[i.value].end(), *catBegin, sameCategory);
324  std::get<i.value>(this->mMaxOffset) = std::distance(this->mGroupedIndices[i.value].begin(), range.second);
325  std::get<i.value>(this->mCurrent).setCursor(range.first->second);
326  });
327  }

330  {
331  constexpr auto k = sizeof...(Ts);
332  for_<k>([&, this](auto i) {
333  std::get<i.value>(this->mCurrent).moveToEnd();
334  });
335  this->mIsEnd = true;
336  }
337 
338  void addOne()
339  {
340  constexpr auto k = sizeof...(Ts);
341  bool modify = true;
342  bool nextCatAvailable = true;
343  for_<k>([&, this](auto i) {
344  if (modify) {
345  constexpr auto curInd = k - i.value - 1;
346  std::get<curInd>(this->mCurrentIndices)++;
347  uint64_t curGroupedInd = std::get<curInd>(this->mCurrentIndices);
348  std::get<curInd>(this->mCurrent).setCursor(this->mGroupedIndices[curInd][curGroupedInd].second);
349 
350  // If we remain within the same category
351  if (std::get<curInd>(this->mCurrentIndices) < std::get<curInd>(this->mMaxOffset)) {
352  for_<i.value>([&, this](auto j) {
353  constexpr auto curJ = k - i.value + j.value;
354  std::get<curJ>(this->mCurrentIndices) = std::get<curJ - 1>(this->mCurrentIndices);
355  uint64_t curGroupedJ = std::get<curJ>(this->mCurrentIndices);
356  std::get<curJ>(this->mCurrent).setCursor(this->mGroupedIndices[curJ][curGroupedJ].second);
357  });
358  modify = false;
359  }
360  }
361  });
362 
363  // No more combinations within this category - move to the next category, if possible
364  if (modify) {
365  for_<k>([&, this](auto m) {
366  if (std::get<m.value>(this->mCurrentIndices) == this->mGroupedIndices[m.value].size()) {
367  nextCatAvailable = false;
368  }
369  });
370  if (nextCatAvailable) {
371  setRanges();
372  }
373  }
374 
375  this->mIsEnd = modify && !nextCatAvailable;
376  }
377 
378  std::array<std::vector<std::pair<uint64_t, uint64_t>>, sizeof...(Ts)> mGroupedIndices;
379  IndicesType mCurrentIndices;
380 };
