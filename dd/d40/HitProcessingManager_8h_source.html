101 inline void HitProcessingManager::sampleCollisionConstituents()
102 {
103  auto getBackgroundRoundRobin = [this]() {
104  static int bgcounter = 0;
105  int numbg = mSimChains[0]->GetEntries();
106  if (bgcounter == numbg) {
107  bgcounter = 0;
108  }
109  return EventPart(0, bgcounter++);
110  };
111 
112  const int nsignalids = mSimChains.size() - 1;
113  auto getSignalRoundRobin = [this, nsignalids]() {
114  static int bgcounter = 0;
115  static int signalid = 0;
116  static std::vector<int> counter(nsignalids, 0);
117  if (signalid == nsignalids) {
118  signalid = 0;
119  }
120  const auto realsourceid = signalid + 1;
121  int numentries = mSimChains[realsourceid]->GetEntries();
122  if (counter[signalid] == numentries) {
123  counter[signalid] = 0;
124  }
125  EventPart e(realsourceid, counter[signalid]);
126  counter[signalid]++;
127  signalid++;
128  return e;
129  };
130 
131  // we fill mDigitizationContext.mEventParts
132  auto& eventparts = mDigitizationContext.getEventParts();
133  eventparts.clear();
134  eventparts.resize(mDigitizationContext.getEventRecords().size());
135  for (int i = 0; i < mDigitizationContext.getEventRecords().size(); ++i) {
136  eventparts[i].clear();
137  // push any number of constituents?
138  // for the moment just 2 : one background and one signal
139  eventparts[i].emplace_back(getBackgroundRoundRobin());
140  if (mSimChains.size() > 1) {
141  eventparts[i].emplace_back(getSignalRoundRobin());
142  }
143  }
144 
145  // push any number of constituents?
146  // for the moment just max 2 : one background and one signal
147  mDigitizationContext.setMaxNumberParts(1);
148  if (mSimChains.size() > 1) {
149  mDigitizationContext.setMaxNumberParts(2);
150  }
151 
152  mDigitizationContext.printCollisionSummary();
153 }
