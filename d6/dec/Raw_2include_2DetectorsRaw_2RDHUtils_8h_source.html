492 inline LinkSubSpec_t RDHUtils::getSubSpec(uint16_t cru, uint8_t link, uint8_t endpoint, uint16_t feeId)
493 {
494  /*
495  // RS Temporarily suppress this way since such a subspec does not define the TOF/TPC links in a unique way
496  // define subspecification as in DataDistribution
497  int linkValue = (LinkSubSpec_t(link) + 1) << (endpoint == 1 ? 8 : 0);
498  return (LinkSubSpec_t(cru) << 16) | linkValue;
499  */
500  // RS Temporarily suppress this way since such a link is ambiguous
501  uint16_t seq[3] = {cru, uint16_t((uint16_t(link) << 8) | endpoint), feeId};
502  return fletcher32(seq, 3);
503 }
